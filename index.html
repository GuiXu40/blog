$${f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}a < \mathop{{x}}\nolimits_{{1}} < \mathop{{x}}\nolimits_{{2}} < \mathop{{x}}\nolimits_{{3}} < ... < x\mathop{{}}\nolimits_{{n}} < b\text{时}\text{，}\text{在} \left[ x\mathop{{}}\nolimits_{{1}},x\mathop{{}}\nolimits_{{n}} \left] \text{内}\text{至}\text{少}\text{存}\text{在}\text{一}\text{点} \xi \text{，}\text{使}\text{得}f \left( \xi \left) =\frac{{f \left( x\mathop{{}}\nolimits_{{1}} \left) +f \left( x\mathop{{}}\nolimits_{{2}} \left) +...+f \left( x\mathop{{}}\nolimits_{{n}} \right) \right. \right. \right. \right. }}{{n}}\right. \right. \right. \right. \right. \right. \right. \right. } $$

$ {\text{设}f \left( x \left) \text{在}x\mathop{{}}\nolimits_{{0}}\text{领}\text{域}\text{上}\text{有}\text{定}\text{义}\text{，}\text{若}x\mathop{{}}\nolimits_{{0}}\text{为}f \left( x \left) \text{的}\text{极}\text{值}\text{点}\text{，}\text{且}f \left( x \left) \text{在}x\mathop{{}}\nolimits_{{0}}\text{处}\text{可}\text{导}\text{，}\text{则}f\mathop{{}}\nolimits^{{\text{'}}} \left( x\mathop{{}}\nolimits_{{0}} \left) =0\right. \right. \right. \right. \right. \right. \right. \right. }$
