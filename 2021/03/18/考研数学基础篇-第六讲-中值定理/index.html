有界与最值定理

$ {f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}\text{ }m \le f \left( x \left) \le M,\text{ }m,M\text{分}\text{别}\text{为}f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{的}\text{最}\text{小}\text{值}\text{与}\text{最}\text{大}\text{值}\text{。}\right. \right. \right. \right. \right. \right. \right. \right. \right. \right. } $

零点存在定理

$ {f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}\text{ }\text{当}f \left( a \left) ⸳f \left( b \left) < 0\text{时}\text{，}\text{存}\text{在} \xi \in \left( a,b \left) \text{，}\text{使}\text{得}f \left( \xi \left) =0\text{。}\right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. }$

介值定理

$ {f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}\text{ }m \le c \le M\text{时}\text{，}\text{存}\text{在} \xi \in \left[ a,b \left] \text{，}\text{使}\text{得}f \left( \xi \left) =c.\right. \right. \right. \right. \right. \right. \right. \right. }$

平均值定理

$${f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}a < \mathop{{x}}\nolimits_{{1}} < \mathop{{x}}\nolimits_{{2}} < \mathop{{x}}\nolimits_{{3}} < ... < x\mathop{{}}\nolimits_{{n}} < b\text{时}\text{，}\text{在} \left[ x\mathop{{}}\nolimits_{{1}},x\mathop{{}}\nolimits_{{n}} \left] \text{内}\text{至}\text{少}\text{存}\text{在}\text{一}\text{点} \xi \text{，}\text{使}\text{得}f \left( \xi \left) =\frac{{f \left( x\mathop{{}}\nolimits_{{1}} \left) +f \left( x\mathop{{}}\nolimits_{{2}} \left) +...+f \left( x\mathop{{}}\nolimits_{{n}} \right) \right. \right. \right. \right. }}{{n}}\right. \right. \right. \right. \right. \right. \right. \right. } $$

费马定理

$ {\text{设}f \left( x \left) \text{在}x\mathop{{}}\nolimits_{{0}}\text{领}\text{域}\text{上}\text{有}\text{定}\text{义}\text{，}\text{若}x\mathop{{}}\nolimits_{{0}}\text{为}f \left( x \left) \text{的}\text{极}\text{值}\text{点}\text{，}\text{且}f \left( x \left) \text{在}x\mathop{{}}\nolimits_{{0}}\text{处}\text{可}\text{导}\text{，}\text{则}f\mathop{{}}\nolimits^{{\text{'}}} \left( x\mathop{{}}\nolimits_{{0}} \left) =0\right. \right. \right. \right. \right. \right. \right. \right. }$

罗尔定理

${\text{若}f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}\text{在}\text{开}\text{区}\text{间} \left( a,b \left) \text{可}\text{导}\text{，}\text{且}f \left( a \left) =f \left( b \left) ,\text{则}\text{存}\text{在} \xi \in \left( a,b \left) \text{，}\text{使}\text{得}f\mathop{{}}\nolimits^{{\text{'}}} \left( \xi \left) =0\right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. } $

拉格朗日中值定理

${\text{若}f \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}\text{在}\text{开}\text{区}\text{间} \left( a,b \left) \text{可}\text{导}\text{，}\text{则} \exists \xi \in \left( a,b \left) \text{使}\text{得}f\mathop{{}}\nolimits^{{\text{'}}} \left( \xi \left) =\frac{{f \left( b \left) -f \left( a \right) \right. \right. }}{{b-a}}\right. \right. \right. \right. \right. \right. \right. \right. \right. \right. } $

$ {\text{若}f \left( x \left) ,g \left( x \left) \text{在} \left[ a,b \left] \text{上}\text{连}\text{续}\text{，}\text{在}\text{开}\text{区}\text{间} \left( a,b \left) \text{可}\text{导}\text{且}g \left( a \left) \neq g \left( b \left) \text{ }g\mathop{{}}\nolimits^{{\text{'}}} \left( x \left) \neq 0\text{，}\text{则} \exists \xi \in \left( a,b \left) \text{使}\text{得}\frac{{f\mathop{{}}\nolimits^{{\text{'}}} \left( \xi \right) }}{{g\text{'} \left( \xi \right) }}=\frac{{f \left( b \left) -f \left( a \right) \right. \right. }}{{g \left( b \left) -g \left( a \right) \right. \right. }}\right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. \right. }$
